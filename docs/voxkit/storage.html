180        os.makedirs(dataset_storage_path, exist_ok=True)
181        
182        # Create metadata
183        from datetime import datetime
184        metadata: DatasetMetadata = {
185            "name": dataset_name,

190            "speaker_count": stats.get("speaker_count", 0),

194        }

210

212        
213    except Exception as e:

def validate_dataset( dataset_path: str, analysis_method: str = 'standard') -> Tuple[bool, str, dict]: 
 45def validate_dataset(
 46    dataset_path: str, analysis_method: str = "standard"
 47) -> Tuple[bool, str, dict]:
 48    """
 49    Validate if a dataset follows Kaldi organization pattern.
 50    
 51    Expected structure:
 52        dataset_root/
 53        ├── speaker_001/
 54        │   ├── audio_001.wav
 55        │   ├── audio_001.lab
 56        │   ├── audio_002.wav
 57        │   └── audio_002.lab
 58        └── speaker_002/
 59            ├── audio_001.wav
 60            └── audio_001.lab
 61    
 62    Args:
 63        dataset_path: Path to dataset root directory
 64        analysis_method: Analysis method - "quick", "standard", or "deep"
 65            - quick: Fast scan with minimal validation
 66            - standard: Full validation with file counting
 67            - deep: Detailed analysis including audio durations and quality checks
 68        
 69    Returns:
 70        Tuple of (is_valid, message, stats_dict)
 71        stats_dict contains: speaker_count, audio_file_count, transcribed
 72        For deep analysis, also includes: total_duration, avg_duration, etc.
 73    """
 74    if not os.path.exists(dataset_path):
 75        return False, "Dataset path does not exist", {}
 76    
 77    if not os.path.isdir(dataset_path):
 78        return False, "Dataset path is not a directory", {}
 79    
 80    # Get all subdirectories (potential speaker directories)
 81    try:
 82        entries = list(os.scandir(dataset_path))
 83        speaker_dirs = [e for e in entries if e.is_dir()]
 84        
 85        if not speaker_dirs:
 86            return False, "No speaker subdirectories found in dataset root", {}
 87        
 88        speaker_count = len(speaker_dirs)
 89        total_audio_files = 0
 90        total_lab_files = 0
 91        audio_extensions = {'.wav', '.flac', '.mp3', '.ogg', '.m4a'}
 92        issues = []
 93        
 94        for speaker_dir in speaker_dirs:
 95            speaker_path = speaker_dir.path
 96            speaker_name = speaker_dir.name
 97            
 98            # Get files in speaker directory
 99            try:
100                files = list(os.scandir(speaker_path))
101                audio_files = [f for f in files if f.is_file() and Path(f.name).suffix.lower() in audio_extensions]
102                lab_files = [f for f in files if f.is_file() and f.name.endswith('.lab')]
103                
104                if not audio_files:
105                    issues.append(f"Speaker '{speaker_name}' has no audio files")
106                    continue
107                
108                total_audio_files += len(audio_files)
109                total_lab_files += len(lab_files)
110                
111                # Check for matching .lab files
112                audio_stems = {Path(f.name).stem for f in audio_files}
113                lab_stems = {Path(f.name).stem for f in lab_files}
114                
115                missing_labs = audio_stems - lab_stems
116                if missing_labs:
117                    issues.append(f"Speaker '{speaker_name}' missing .lab files for: {', '.join(list(missing_labs)[:3])}{' ...' if len(missing_labs) > 3 else ''}")
118                    
119            except PermissionError:
120                issues.append(f"Permission denied accessing speaker directory '{speaker_name}'")
121                continue
122        
123        # Determine if dataset is transcribed
124        transcribed = total_lab_files > 0 and total_lab_files >= (total_audio_files * 0.5)  # At least 50% transcribed
125        
126        stats = {
127            "speaker_count": speaker_count,
128            "audio_file_count": total_audio_files,
129            "transcribed": transcribed
130        }
131        
132        if total_audio_files == 0:
133            return False, "No audio files found in any speaker directory", stats
134        
135        if issues:
136            issue_msg = "Dataset is partially valid. Issues found:\n" + "\n".join(f"- {issue}" for issue in issues[:5])
137            if len(issues) > 5:
138                issue_msg += f"\n... and {len(issues) - 5} more issues"
139            return True, issue_msg, stats  # Still valid but with warnings
140        
141        return True, f"Dataset is valid: {speaker_count} speakers, {total_audio_files} audio files, {'transcribed' if transcribed else 'not transcribed'}", stats
142        
143    except Exception as e:
144        return False, f"Error validating dataset: {str(e)}", {}
